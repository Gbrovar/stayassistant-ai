export function rankPlaces(places, userProfile, memory) {

  return places
    .map(place => {

      let score = 0

      // ⭐ rating base
      score += (place.rating || 4) * 2

      // 📍 distancia (más cerca = mejor)
      if (place.distance) {
        score += Math.max(0, 2000 - place.distance) / 500
      }

      // 🧠 preferencias
      if (userProfile.preferences.includes("vegan") &&
          place.name.toLowerCase().includes("vegan")) {
        score += 3
      }

      if (memory.budget === "low") {
        score += 1 // simple heuristic
      }

      // 🔥 intereses
      if (userProfile.interests.includes("food")) {
        score += 2
      }

      return {
        ...place,
        score
      }

    })
    .sort((a, b) => b.score - a.score)
}
